def createVocabList(dataSet):
    #
    vocabSet = set([])      
    #
    for document in dataSet:      
        vocabSet = vocabSet | set(document)
    
    return list(vocabSet)

def bagOfWords2Vec(vocabList, inputSet):
    #0
    returnVec = [0] * len(vocabList)
    for word in inputSet:
        if word in vocabList:
            returnVec[vocabList.index(word)] += 1
        else:
            print("The Word [%d] is not in my Vocabulary!" % word)
    
    return returnVec

def trainNB0(trainMatrix, trainCategory):    
    #
    numTrainDocs = len(trainMatrix)
    #
    numWords = len(trainMatrix[0])
    
    #
    #label == 1 abusive  p(c1)
    # p(c0) = 1 - p(c1) 
    pAbusive = sum(trainCategory) / numTrainDocs
    
    #
    #label == 0 & label == 1 conditional probability p(wi|c0)  p(wi|c1)
    #
    #is abusive or not document p1Num or p0Num +1
    # +1
    #
    p0Num = np.ones(numWords)
    p1Num = np.ones(numWords)
    p0Denom = 2.0 
    p1Denom = 2.0   
    for i in range(numTrainDocs):
        #label == 1
        if trainCategory[i] == 1:
            p1Num += trainMatrix[i]          #
            p1Denom += sum(trainMatrix[i])   #label==1
        else:
            p0Num += trainMatrix[i]          #
            p0Denom += sum(trainMatrix[i])
    
    print("p0Num:\n", p0Num)
    print("p0Denom:\n", p0Denom)
    print("\np1Num:\n", p1Num)
    print("p1Denom:\n", p1Denom)
    print("\n-----------------------------------------------\n")
    
    #
    # p(wi|ci)  log 
    p1Vect = np.log(p1Num/p1Denom)              #p(W|c1)  
    p0Vect = np.log(p0Num/p0Denom)              #p(W|c0)
    
    return p0Vect, p1Vect, pAbusive

实现朴素贝叶斯分类器¶

$\textbf{w}$ $\textbf{new_w}$ $p(c_i|\textbf{new_w})$$p(c_0|\textbf{new_w})$ $p(c_1|\textbf{new_w})$

trainNB0 function log $p(\textbf{w}|c_i)$ = $p(w_0,w_1,...,w_N|c_i)$ = $p(w_0|c_i)p(w_1|c_i)...p(w_N|c_i)$ $log(p(w_0|c_i)) + log(p(w_1|c_i)) + ... + log(p(w_N|c_i))$

def classifyNB(vec2Classify, p0Vec, p1Vec, pClass1):
    #log
    # c_i  c1  c0 
    #
    p1 = sum(vec2Classify * p1Vec) + np.log(pClass1)    
    p0 = sum(vec2Classify * p0Vec) + np.log(1.0 - pClass1)
    #
    if p1 > p0:
        return 1
    else: 
        return 0
